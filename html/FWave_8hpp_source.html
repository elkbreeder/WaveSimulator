315  if(hLeft < dryTol && hRight < dryTol) { //both cells are dry
316  wetDryState = DryDry;

354  void computeWaveDecomposition( const T i_waveSpeeds[2],
355  T o_fWaves[2][2]
356  ) const {
357  //TODO: Update documentation.
358  //Eigenvalues***********************************************************************************************
359  //Computed somewhere before.
360  //An option would be to use the char. speeds:
361  //
362  //lambda^1 = u_{i-1} - sqrt(g*h_{i-1})
363  //lambda^2 = u_i + sqrt(g*h_i)
364  //Matrix of right eigenvectors******************************************************************************
365  // 1 1
366  // R =
367  // u_{i-1} - sqrt(g * h_{i-1}) u_i + sqrt(g * h_i)
368  //**********************************************************************************************************
369  // u_i + sqrt(g * h_i) -1
370  // R^{-1} = 1 / (u_i - sqrt(g * h_i) - u_{i-1} + sqrt(g * h_{i-1}) *
371  // -( u_{i-1} - sqrt(g * h_{i-1}) ) 1
372  //**********************************************************************************************************
373  // hu
374  // f(q) =
375  // hu^2 + 1/2 g * h^2
376  //
377  //**********************************************************************************************************
378  // 0
379  // \delta x \Psi =
380  // -g * 1/2 * (h_i + h_{i-1}) * (b_i - b_{i+1})
381  //**********************************************************************************************************
382  // beta = R^{-1} * (f(Q_i) - f(Q_{i-1}) - \delta x \Psi)
383  //**********************************************************************************************************
384 
385  //assert: wave speed of the 1st wave family should be less than the speed of the 2nd wave family.
386  assert( i_waveSpeeds[0] < i_waveSpeeds[1] );
387 
388  T lambdaDif = i_waveSpeeds[1] - i_waveSpeeds[0];
389 
390  //assert: no division by zero
391  assert(std::abs(lambdaDif) > zeroTol);
392 
393  //compute the inverse matrix R^{-1}
394  T Rinv[2][2];
395 
396  T oneDivLambdaDif = (T)1. / lambdaDif;
397  Rinv[0][0] = oneDivLambdaDif * i_waveSpeeds[1];
398  Rinv[0][1] = -oneDivLambdaDif;
399 
400  Rinv[1][0] = oneDivLambdaDif * -i_waveSpeeds[0];
401  Rinv[1][1] = oneDivLambdaDif;
402 
403  //right hand side
404  T fDif[2];
405 
406  //calculate modified (bathymetry!) flux difference
407  // f(Q_i) - f(Q_{i-1})
408  fDif[0] = huRight - huLeft;
409  fDif[1] = huRight * uRight + (T).5 * g * hRight * hRight
410  -(huLeft * uLeft + (T).5 * g * hLeft * hLeft);
411 
412  // \delta x \Psi[2]
413  T psi = -g * (T).5 * (hRight + hLeft)*(bRight - bLeft);
414  fDif[1] -= psi;
415 
416  //solve linear equations
417  T beta[2];
418  beta[0] = Rinv[0][0] * fDif[0] + Rinv[0][1] * fDif[1];
419  beta[1] = Rinv[1][0] * fDif[0] + Rinv[1][1] * fDif[1];
420 
421  //return f-waves
422  o_fWaves[0][0] = beta[0];
423  o_fWaves[0][1] = beta[0] * i_waveSpeeds[0];
424 
425  o_fWaves[1][0] = beta[1];
426  o_fWaves[1][1] = beta[1] * i_waveSpeeds[1];
427  }

434  void computeWaveSpeeds( T o_waveSpeeds[2]) const {
435  //compute eigenvalues of the jacobian matrices in states Q_{i-1} and Q_{i}
436  T characteristicSpeeds[2];
437  characteristicSpeeds[0] = uLeft - std::sqrt(g*hLeft);
438  characteristicSpeeds[1] = uRight + std::sqrt(g*hRight);
439 
440  //compute "Roe speeds"
441  T hRoe = (T).5 * (hRight + hLeft);
442  T uRoe = uLeft * std::sqrt(hLeft) + uRight * std::sqrt(hRight);
443  uRoe /= std::sqrt(hLeft) + std::sqrt(hRight);
444 
445  T roeSpeeds[2];
446  roeSpeeds[0] = uRoe - std::sqrt(g*hRoe);
447  roeSpeeds[1] = uRoe + std::sqrt(g*hRoe);
448 
449  //computer eindfeldt speeds
450  T einfeldtSpeeds[2];
451  einfeldtSpeeds[0] = std::min(characteristicSpeeds[0], roeSpeeds[0]);
452  einfeldtSpeeds[1] = std::max(characteristicSpeeds[1], roeSpeeds[1]);
453 
454  //set wave speeds
455  o_waveSpeeds[0] = einfeldtSpeeds[0];
456  o_waveSpeeds[1] = einfeldtSpeeds[1];
457  }
