490 void SWE_Block::computeMaxTimestep( const float i_dryTol,
491  const float i_cflNumber ) {
492 
493  // initialize the maximum wave speed
494  float l_maximumWaveSpeed = (float) 0;
495 
496  // compute the maximum wave speed within the grid
497  for(int i=1; i <= nx; i++) {
498  for(int j=1; j <= ny; j++) {
499  if( h[i][j] > i_dryTol ) {
500  float l_momentum = std::max( std::abs( hu[i][j] ),
501  std::abs( hv[i][j] ) );
502 
503  float l_particleVelocity = l_momentum / h[i][j];
504 
505  // approximate the wave speed
506  float l_waveSpeed = l_particleVelocity + std::sqrt( g * h[i][j] );
507 
508  l_maximumWaveSpeed = std::max( l_maximumWaveSpeed, l_waveSpeed );
509  }
510  }
511  }
512 
513  float l_minimumCellLength = std::min( dx, dy );
514 
515  // set the maximum time step variable
516  maxTimestep = l_minimumCellLength / l_maximumWaveSpeed;
517 
518  // apply the CFL condition
519  maxTimestep *= i_cflNumber;
520 }

535 
536  // CONNECT boundary conditions are set in the calling function setGhostLayer
537  // PASSIVE boundary conditions need to be set by the component using SWE_Block
538 
539  // left boundary
540  switch(boundary[BND_LEFT]) {
541  case WALL:
542  {
543  for(int j=1; j<=ny; j++) {
544  h[0][j] = h[1][j];
545  hu[0][j] = -hu[1][j];
546  hv[0][j] = hv[1][j];
547  };
548  break;
549  }
550  case OUTFLOW:
551  {
552  for(int j=1; j<=ny; j++) {
553  h[0][j] = h[1][j];
554  hu[0][j] = hu[1][j];
555  hv[0][j] = hv[1][j];
556  };
557  break;
558  }
559  case CONNECT:
560  case PASSIVE:
561  break;
562  default:
563  assert(false);
564  break;
565  };
566 
567  // right boundary
568  switch(boundary[BND_RIGHT]) {
569  case WALL:
570  {

572  h[nx+1][j] = h[nx][j];
573  hu[nx+1][j] = -hu[nx][j];
574  hv[nx+1][j] = hv[nx][j];
575  };

577  }

582  hu[nx+1][j] = hu[nx][j];

584  };
585  break;
586  }
587  case CONNECT:
588  case PASSIVE:
589  break;

593  };

651  /*
652  * Set values in corner ghost cells. Required for dimensional splitting and visualizuation.
653  * The quantities in the corner ghost cells are chosen to generate a zero Riemann solutions
654  * (steady state) with the neighboring cells. For the lower left corner (0,0) using
655  * the values of (1,1) generates a steady state (zero) Riemann problem for (0,0) - (0,1) and
656  * (0,0) - (1,0) for both outflow and reflecting boundary conditions.
657  *
658  * Remark: Unsplit methods don't need corner values.
659  *
660  * Sketch (reflecting boundary conditions, lower left corner):
661  * <pre>
662  * **************************
663  * * _ _ * _ _ *
664  * Ghost * | h | * | h | *
665  * cell ------> * | -hu | * | hu | * <------ Cell (1,1) inside the domain
666  * (0,1) * |_ hv _| * |_ hv _| *
667  * * * *
668  * **************************
669  * * _ _ * _ _ *
670  * Corner Ghost * | h | * | h | *
671  * cell ------> * | hu | * | hu | * <----- Ghost cell (1,0)
672  * (0,0) * |_ hv _| * |_-hv _| *
673  * * * *
674  * **************************
675  * </pre>
676  */

678  hu[0][0] = hu[1][1];
679  hv[0][0] = hv[1][1];
680 
681  h [0][ny+1] = h [1][ny];
682  hu[0][ny+1] = hu[1][ny];
683  hv[0][ny+1] = hv[1][ny];
684 
685  h [nx+1][0] = h [nx][1];
